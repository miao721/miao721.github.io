urlforum/{id}
调用方式GET
名称主题列表
示例urlwww.nmmnm.co/api/forum/5
必备参数id
返回示例 {
"id": 21,//主题帖ID
"title": "测试一下",//标题
"username": "= =",//昵称
"fid": 2,//版块名
"tid": 1,//版块内帖子id，app内此字段应无效化
"content": "看看",//内容
"status": 0,//状态，1为锁定
"reply_count": 1,//回复数量
"lastpost": "1532575830",//最后回复时间
"created_at": "2018-07-26 07:25:10",//主题发布时间
"digest": 0,//是否为精华，非0值均为精华贴
"type": 0,//1为套红，2为加星
"photofile": "",//附件地址注意:该字段已被废弃，可以不必理会
"top": 0,//置顶权重，值越大，帖子显示越靠前
"bbsbigimage": 1//头像id(1-40)
},
